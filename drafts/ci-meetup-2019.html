.gitlab-ci.yaml

stages:
    - build
    - test
build:
    services:
        - docker:dind # docker in docker 
    script:
        - docker build ...
        - docker login ...
        - docker push ...
    only:
        changes:
            - Dockerfile
            - src/*
            - html/*
test:
    image: docker:stable
    stage: test
    services:
        - docker:dind # docker in docker 
    script:
        - docker login
        - xxx
    coverage: 'xxxx'
  • Stages can be set to run in parallel!
workon demo
